# Fix Key Vault Soft Delete + Purge Protection (correct syntax)
$kvNames = @("kv-pyx-prod","kv-pyx-test","kv-pyx-qa","kv-extracts-prod",
             "kv-pyx-stage","kv-extracts-preprod","kv-target-pop-prod",
             "kv-target-pop-preprod","kv-datasystem-prod","kv-datasystem-antprod",
             "kv-datasystem-test","kv-datasystem-qa","kv-backend-automation")

foreach ($kv in $kvNames) {
    try {
        $vault = Get-AzKeyVault -VaultName $kv -ErrorAction Stop
        
        # Correct syntax — no positional 'True', use switch params
        Update-AzKeyVault -VaultName $kv `
            -ResourceGroupName $vault.ResourceGroupName `
            -EnableSoftDelete `
            -EnablePurgeProtection
        
        Write-Host "[FIXED] $kv" -ForegroundColor Green
    } catch {
        Write-Host "[FAILED] $kv - $_" -ForegroundColor Red
    }
}
