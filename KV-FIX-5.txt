    TotalResources      = 672

    SQLMissingATP       = 17
    StorageNoHTTPS      = 0
    StorageNoTLS        = 1
    StoragePublicAccess = 44
    KVNoSoftDelete      = 0
    KVNoPurgeProtection = 32
    AppServicesNoHTTPS  = 23

$after = @{

    UnhealthyResources  = 0
    SQLMissingTDE       = 0
    SQLMissingATP       = 0
    StorageNoHTTPS      = 0
    StorageNoTLS        = 0
    StoragePublicAccess = 0
    KVNoSoftDelete      = 0
    KVNoPurgeProtection = 0
    AppServicesNoHTTPS  = 0
}

$detailRows = ""
$fixedRows = ""
$manualRows = ""
$totalFixed = 0
$totalFailed = 0

foreach ($sub in Get-AzSubscription) {
    Set-AzContext -SubscriptionId $sub.Id | Out-Null
    Write-Host "[SUB] $($sub.Name)" -ForegroundColor Cyan

    # KEY VAULTS
    foreach ($v in Get-AzKeyVault) {
        $kv = Get-AzKeyVault -VaultName $v.VaultName -ErrorAction SilentlyContinue
        $purge = $kv.EnablePurgeProtection
        $soft = $kv.EnableSoftDelete

        if ($purge -ne $true) { $after.KVNoPurgeProtection++ }
        if ($soft -ne $true)  { $after.KVNoSoftDelete++ }

        $status = if($purge -eq $true){"COMPLIANT ✅"}else{"NON-COMPLIANT ❌"}
        $color  = if($purge -eq $true){"#155724"}else{"#721c24"}
        $bg     = if($purge -eq $true){"#d4edda"}else{"#f8d7da"}

        $detailRows += "<tr>
            <td>$($v.VaultName)</td>
            <td>$($sub.Name)</td>
            <td>Key Vault</td>
            <td>Soft Delete=$soft | Purge Protection=$purge</td>
            <td style='background:$bg;color:$color;font-weight:bold;text-align:center'>$status</td>
        </tr>"

        if($purge -eq $true){
            $totalFixed++
            Write-Host "  [OK] $($v.VaultName) - Purge=True" -ForegroundColor Green
        }
    }

    # STORAGE ACCOUNTS
    foreach ($sa in Get-AzStorageAccount) {
        $https = $sa.EnableHttpsTrafficOnly
        $pub   = $sa.AllowBlobPublicAccess
        try {
            $blob = (Get-AzStorageAccount -ResourceGroupName $sa.ResourceGroupName -Name $sa.StorageAccountName).BlobServiceProperties.DeleteRetentionPolicy.Enabled
        } catch { $blob = $false }

        if (-not $https)           { $after.StorageNoHTTPS++ }
        if ($pub -eq $true)        { $after.StoragePublicAccess++ }
        if ($blob -ne $true)       { $after.KVNoSoftDelete++ }

        $allGood = $https -and ($pub -ne $true)
        $status  = if($allGood){"COMPLIANT ✅"}else{"NON-COMPLIANT ❌"}
        $color   = if($allGood){"#155724"}else{"#721c24"}
        $bg      = if($allGood){"#d4edda"}else{"#f8d7da"}

        $detailRows += "<tr>
            <td>$($sa.StorageAccountName)</td>
            <td>$($sub.Name)</td>
            <td>Storage Account</td>
            <td>HTTPS=$https | PublicAccess=$pub | SoftDelete=$blob</td>
            <td style='background:$bg;color:$color;font-weight:bold;text-align:center'>$status</td>
        </tr>"

        if($allGood){ $totalFixed++ }
    }

    # APP SERVICES
    foreach ($app in Get-AzWebApp) {
        $https = $app.HttpsOnly
        if (-not $https) { $after.AppServicesNoHTTPS++ }

        $status = if($https){"COMPLIANT ✅"}else{"NON-COMPLIANT ❌"}
        $color  = if($https){"#155724"}else{"#721c24"}
        $bg     = if($https){"#d4edda"}else{"#f8d7da"}

        $detailRows += "<tr>
            <td>$($app.Name)</td>
            <td>$($sub.Name)</td>
            <td>App Service</td>
            <td>HTTPS Only=$https</td>
            <td style='background:$bg;color:$color;font-weight:bold;text-align:center'>$status</td>
        </tr>"

        if($https){ $totalFixed++ }
    }

    # SQL SERVERS
    foreach ($sql in Get-AzSqlServer) {
        try {
            $atp = Get-AzSqlServerAdvancedThreatProtectionSetting `
                -ServerName $sql.ServerName `
                -ResourceGroupName $sql.ResourceGroupName `
                -ErrorAction Stop
            $atpOk = $atp.ThreatDetectionState -eq "Enabled"
        } catch { $atpOk = $false }

        if (-not $atpOk) { $after.SQLMissingATP++ }

        $status = if($atpOk){"COMPLIANT ✅"}else{"NON-COMPLIANT ❌"}
        $color  = if($atpOk){"#155724"}else{"#721c24"}
        $bg     = if($atpOk){"#d4edda"}else{"#f8d7da"}

        $detailRows += "<tr>
            <td>$($sql.ServerName)</td>
            <td>$($sub.Name)</td>
            <td>SQL Server</td>
            <td>ATP=$($atp.ThreatDetectionState)</td>
            <td style='background:$bg;color:$color;font-weight:bold;text-align:center'>$status</td>
        </tr>"

        if($atpOk){ $totalFixed++ }
    }
}

# CALCULATE TOTALS
$after.UnhealthyResources = $after.SQLMissingATP + $after.StoragePublicAccess + $after.KVNoPurgeProtection + $after.AppServicesNoHTTPS
$totalIssuesFixed = $before.UnhealthyResources - $after.UnhealthyResources
$improvementPct = [math]::Round(($totalIssuesFixed / $before.UnhealthyResources) * 100)
